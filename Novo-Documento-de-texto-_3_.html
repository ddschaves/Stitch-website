// Funo de contagem regressiva
const countdownDate = new Date("March 8, 2025 16:00:00 UTC").getTime();
const timerElement = document.getElementById('timer');
function updateCountdown() {
const now = new Date().getTime();
const distance = countdownDate - now;
if (distance <= 0) {
timerElement.innerHTML = "A pr-venda terminou!";
clearInterval(countdown);
return;
}
const days = Math.floor(distance / (1000 * 60 * 60 * 24));
const hours = Math.floor((distance % (1000 * 60 * 60 * 24)) / (1000 *
60 * 60));
const minutes = Math.floor((distance % (1000 * 60 * 60)) / (1000 *
60));
const seconds = Math.floor((distance % (1000 * 60)) / 1000);
timerElement.innerHTML = `${days}d ${hours}h ${minutes}m
${seconds}s`;
}
const countdown = setInterval(updateCountdown, 1000);
